foxhole_soviet_molotov_claims_category = {
    foxhole_demand_estonia = {
        icon = border_war
        cost = 25
        fire_only_once = yes

        visible = {
            any_state = {
                is_core_of = EST
                is_claimed_by = SOV
                NOT = {
                    is_owned_by = SOV
                }
            }
        }

        days_remove = 25

        cancel_trigger = {
            hidden_trigger = {
                NOT = {
                    any_other_country = {
                        any_owned_state = {
                            is_core_of = EST
                            is_claimed_by = SOV
                            NOT = {
                                is_owned_by = SOV
                            }
                        }
                    }
                }
            }
        }

        remove_effect = {
            every_other_country = {
                limit = {
                    any_owned_state = {
                        is_core_of = EST
                        is_claimed_by = SOV
                    }
                }
                SOV = {
                    create_wargoal = { #take_claimed_state would be better but setting the target state would be a pain
                        type = annex_everything
                        target = PREV
                    }
                }
            }
        }

        complete_effect = {
            every_other_country = {
                limit = {
                    any_owned_state = {
                        is_core_of = EST
                        is_claimed_by = SOV
                    }
                }
                country_event = {
                    id = foxhole_SOV_demands.0
                    days = 1
                }
            }
        }
    }
